def forcenteral(f):
    f.update()
    fw = f.winfo_width()
    fh = f.winfo_height()
    sw = f.winfo_screenwidth()
    sh = f.winfo_screenheight()
    x = ( sw - fw )/2
    y = ( sh - fh )/2
    f.geometry( "%dx%d+%d+%d" %(fw , fh , x , y) )
